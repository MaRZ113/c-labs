#include <stdio.h>   // Подключаем стандартную библиотеку ввода/вывода (для printf, scanf, работы с файлами)
#include <stdlib.h>  // Подключаем библиотеку для выделения памяти (malloc, free) и генерации чисел (rand, srand)
#include <time.h>    // Подключаем библиотеку для работы со временем (clock, time, mktime)

int** create_matrix(int m) { // Функция для создания и заполнения матрицы размером m x m
    int i, j; // Объявляем счетчики для циклов
    int **mat = (int**)malloc(m * sizeof(int*)); // Выделяем память под массив указателей на строки матрицы
    for (i = 0; i < m; i++) { // Проходимся циклом по количеству строк
        mat[i] = (int*)malloc(m * sizeof(int)); // Выделяем память под саму строку (столбцы)
        for (j = 0; j < m; j++) { // Проходимся по каждому элементу в строке
            mat[i][j] = rand() % 10; // Записываем в ячейку случайное число от 0 до 9
        }
    }
    return mat; // Возвращаем готовую матрицу
}

void free_matrix(int **mat, int m) { // Функция для очистки оперативной памяти от матрицы
    int i; // Объявляем счетчик
    if (!mat) return; // Если матрица пустая (NULL), сразу выходим, чтобы не было ошибки
    for (i = 0; i < m; i++) { // Пробегаемся по всем строкам
        free(mat[i]); // Освобождаем память из-под каждой строки
    }
    free(mat); // Освобождаем память из-под основного массива указателей
}

int matrix_sum(int **mat, int m) { // Функция подсчета суммы всех элементов матрицы
    int i, j, sum = 0; // Объявляем счетчики и переменную-сумматор (обязательно равную 0)
    for (i = 0; i < m; i++) { // Цикл по строкам
        for (j = 0; j < m; j++) { // Вложенный цикл по столбцам
            sum += mat[i][j]; // Прибавляем значение текущей ячейки к общей сумме
        }
    }
    return sum; // Возвращаем итоговую сумму
}

void print_matrix(FILE *f, int **mat, int m) { // Функция для красивого вывода матрицы в файл
    int i, j; // Счетчики
    for (i = 0; i < m; i++) { // Цикл по строкам
        for (j = 0; j < m; j++) { // Цикл по столбцам
            fprintf(f, "%4d", mat[i][j]); // Печатаем число, выделяя под него 4 символа (для ровных колонок)
        }
        fprintf(f, "\n"); // После вывода всех столбцов строки делаем перенос на новую строку
    }
}

int main(void) { // Главная функция, точка входа в программу
    int d = 0, mon = 0, y = 0, m, tries = 0; // Переменные для даты, размера матрицы и счетчика попыток
    int sum1, sum2; // Переменные для хранения сумм двух матриц
    int **first = NULL, **second = NULL; // Указатели на первую и вторую матрицы (пока пустые)
    FILE *fin, *fout; // Указатели на файлы для чтения и записи
    time_t now; // Переменная для хранения текущего времени в секундах
    struct tm target_date = {0}; // Структура для хранения целевой даты (обнуляем ее)
    double diff_seconds; // Переменная для разницы во времени в секундах
    long diff_days; // Переменная для разницы во времени в днях
    clock_t start_time, end_time; // Переменные для замера времени выполнения программы

    start_time = clock(); // Запоминаем время начала работы программы
    srand((unsigned int)time(NULL)); // Инициализируем генератор случайностей текущим временем

    fin = fopen("input.txt", "r"); // Открываем файл input.txt для чтения ("r" - read)
    if (!fin) { // Если файл не найден (вернулся NULL)
        printf("Error: input.txt not found.\n"); // Выводим сообщение об ошибке в консоль
        return 1; // Завершаем программу с кодом ошибки 1
    }
    
    if (fscanf(fin, "%d%*c%d%*c%d", &d, &mon, &y) != 3) { // Читаем 3 числа, игнорируя любые разделители (%*c)
        printf("Error: wrong date format in file.\n"); // Если считалось не 3 числа, выводим ошибку
        fclose(fin); // Закрываем файл
        return 1; // Завершаем программу
    }
    fclose(fin); // Если всё хорошо, просто закрываем файл после чтения

    time(&now); // Записываем в переменную now текущее системное время
    target_date.tm_mday = d; // Записываем считанный день в структуру даты
    target_date.tm_mon = mon - 1; // Записываем месяц (отнимаем 1, т.к. в Си месяцы идут от 0 до 11)
    target_date.tm_year = y - 1900; // Записываем год (отнимаем 1900 по правилам стандарта Си)
    target_date.tm_isdst = -1; // Разрешаем системе самой определить летнее/зимнее время

    diff_seconds = difftime(mktime(&target_date), now); // Превращаем дату в секунды и находим разницу с текущим временем
    diff_days = (long)(diff_seconds / 86400); // Делим разницу в секундах на 86400 (кол-во секунд в сутках), получаем дни
    if (diff_days < 0) diff_days = 0; // Если дата уже прошла, ставим 0 дней, чтобы не выводить минус

    printf("Target date: %02d.%02d.%04d\n", d, mon, y); // Печатаем дату в консоль с ведущими нулями (%02d)
    printf("Days left: %ld\n\n", diff_days); // Печатаем оставшиеся дни

    printf("Enter matrix size M: "); // Просим пользователя ввести размер матрицы
    if (scanf("%d", &m) != 1 || m < 1) return 1; // Считываем M. Если ввели не число или число < 1, выходим

    first = create_matrix(m); // Создаем первую матрицу
    sum1 = matrix_sum(first, m); // Сразу считаем её сумму

    do { // Начинаем цикл поиска (сначала делаем, потом проверяем условие)
        free_matrix(second, m); // Очищаем память от второй матрицы (важно для 2-й и последующих попыток)
        second = create_matrix(m); // Создаем вторую матрицу заново
        sum2 = matrix_sum(second, m); // Считаем сумму второй матрицы
        tries++; // Увеличиваем счетчик попыток на 1
    } while (sum1 != sum2); // Крутим цикл, пока суммы не станут равными

    fout = fopen("output.txt", "w"); // Открываем файл output.txt для записи ("w" - write)
    if (!fout) return 1; // Если файл не создался (например, нет прав), выходим

    fprintf(fout, "Target date: %02d.%02d.%04d\n", d, mon, y); // Пишем целевую дату в файл
    fprintf(fout, "Days left: %ld\n\n", diff_days); // Пишем оставшиеся дни в файл
    
    fprintf(fout, "Matrix size: %dx%d. Equal sum: %d\n", m, m, sum1); // Пишем размер и найденную одинаковую сумму
    fprintf(fout, "Tries (second matrix recreations): %d\n\n", tries); // Пишем количество потраченных попыток

    fprintf(fout, "First matrix:\n"); // Заголовок для первой матрицы
    print_matrix(fout, first, m); // Вызываем функцию печати первой матрицы в файл

    fprintf(fout, "\nSecond matrix:\n"); // Заголовок для второй матрицы
    print_matrix(fout, second, m); // Вызываем функцию печати второй матрицы в файл

    end_time = clock(); // Запоминаем время окончания работы программы
    fprintf(fout, "\nExecution time: %.3f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC); // Считаем и пишем время
    fclose(fout); // Закрываем файл с результатами

    free_matrix(first, m); // Очищаем память из-под первой матрицы
    free_matrix(second, m); // Очищаем память из-под второй матрицы

    printf("Success! Match found. Data saved to output.txt\n"); // Выводим сообщение об успехе в консоль
    return 0; // Штатно завершаем программу
}

#include <stdio.h>   // Библиотека для ввода/вывода (консоль, файлы)
#include <stdlib.h>  // Библиотека для работы с памятью и рандомом
#include <time.h>    // Библиотека для замеров времени

int** create_matrix(int n) { // Функция создания квадратной матрицы N x N
    int i, j; // Счетчики
    int **mat = (int**)malloc(n * sizeof(int*)); // Выделяем память под массив строк
    for (i = 0; i < n; i++) { // Идем по строкам
        mat[i] = (int*)malloc(n * sizeof(int)); // Выделяем память под столбцы
        for (j = 0; j < n; j++) { // Идем по столбцам
            mat[i][j] = rand() % 10; // Генерируем случайное число от 0 до 9
        }
    }
    return mat; // Возвращаем указатель на матрицу
}

void free_matrix(int **mat, int n) { // Очистка памяти
    int i; // Счетчик
    if (!mat) return; // Защита от пустой матрицы
    for (i = 0; i < n; i++) { // Идем по строкам
        free(mat[i]); // Удаляем каждую строку
    }
    free(mat); // Удаляем основной массив
}

void add_matrices(int **a, int **b, int **res, int n) { // Функция сложения матриц
    int i, j; // Счетчики
    for (i = 0; i < n; i++) { // Перебираем строки
        for (j = 0; j < n; j++) { // Перебираем столбцы
            res[i][j] = a[i][j] + b[i][j]; // Складываем соответствующие ячейки
        }
    }
}

void mult_matrices(int **a, int **b, int **res, int n) { // Функция умножения матриц
    int i, j, k; // Три счетчика для алгоритма "строка на столбец"
    for (i = 0; i < n; i++) { // Цикл по строкам первой матрицы
        for (j = 0; j < n; j++) { // Цикл по столбцам второй матрицы
            res[i][j] = 0; // Обязательно обнуляем ячейку результата перед подсчетом
            for (k = 0; k < n; k++) { // Цикл для перемножения элементов
                res[i][j] += a[i][k] * b[k][j]; // Накапливаем сумму произведений
            }
        }
    }
}

void swap_pairs(int **mat, int n) { // Функция замены соседних пар местами
    int i, r1, c1, r2, c2, temp; // Переменные для индексов и временная переменная
    int total = n * n; // Общее количество элементов в матрице
    for (i = 0; i < total - 1; i += 2) { // Идем по матрице с шагом 2 (перебираем парами)
        r1 = i / n; // Вычисляем индекс строки первого элемента (целочисленное деление)
        c1 = i % n; // Вычисляем индекс столбца первого элемента (остаток от деления)
        r2 = (i + 1) / n; // Индекс строки второго (соседнего) элемента
        c2 = (i + 1) % n; // Индекс столбца второго (соседнего) элемента
        
        temp = mat[r1][c1]; // Сохраняем первый элемент во временную переменную
        mat[r1][c1] = mat[r2][c2]; // Перезаписываем первый элемент вторым
        mat[r2][c2] = temp; // Перезаписываем второй элемент сохраненным первым
    }
}

int has_identical_lines(int **mat, int n) { // Функция поиска одинаковых линий (все цифры равны)
    int i, j, ok; // Счетчики и флаг совпадения
    for (i = 0; i < n; i++) { // Проверка всех строк
        ok = 1; // Предполагаем, что линия одинаковая (флаг = 1)
        for (j = 1; j < n; j++) { // Идем по элементам строки, начиная со второго
            if (mat[i][j] != mat[i][0]) { // Если текущий элемент не равен первому элементу строки
                ok = 0; // Сбрасываем флаг (линия не одинаковая)
                break; // Досрочно прерываем внутренний цикл
            }
        }
        if (ok) return 1; // Если флаг остался 1, значит нашли одинаковую строку, возвращаем истину
    }
    
    for (j = 0; j < n; j++) { // Проверка всех столбцов (логика аналогична строкам)
        ok = 1; // Поднимаем флаг
        for (i = 1; i < n; i++) { // Идем по элементам столбца
            if (mat[i][j] != mat[0][j]) { // Сравниваем с первым элементом столбца
                ok = 0; // Сбрасываем флаг при несовпадении
                break; // Выходим из цикла
            }
        }
        if (ok) return 1; // Если столбец одинаковый, возвращаем истину
    }
    return 0; // Если ничего не нашли, возвращаем 0 (ложь)
}

void print_matrix(FILE *f, int **mat, int n) { // Вывод матрицы в файл
    int i, j; // Счетчики
    for (i = 0; i < n; i++) { // Строки
        for (j = 0; j < n; j++) { // Столбцы
            fprintf(f, "%4d", mat[i][j]); // Печатаем элемент с шириной 4 символа
        }
        fprintf(f, "\n"); // Перенос строки
    }
}

    f = fopen("input.txt", "r"); // Открываем файл теперь уже для чтения
    if (!f) return 1; // Проверка открытия
    fscanf(f, "%d", &n); // Читаем размер
    
    a = (int**)malloc(n * sizeof(int*)); // Заново выделяем память под "костяк" A
    b = (int**)malloc(n * sizeof(int*)); // Заново выделяем память под "костяк" B
    for (i = 0; i < n; i++) { // Выделяем память под строки
        a[i] = (int*)malloc(n * sizeof(int));
        b[i] = (int*)malloc(n * sizeof(int));
    }
    
    for (i = 0; i < n; i++) { // Читаем элементы матрицы A из файла
        for (j = 0; j < n; j++) fscanf(f, "%d", &a[i][j]);
    }
    for (i = 0; i < n; i++) { // Читаем элементы матрицы B из файла
        for (j = 0; j < n; j++) fscanf(f, "%d", &b[i][j]);
    }
    fclose(f); // Закрываем файл чтения

    sum = create_matrix(n); // Выделяем память под матрицу-сумму
    prod = create_matrix(n); // Выделяем память под матрицу-произведение
    
    add_matrices(a, b, sum, n); // Вызываем функцию сложения
    mult_matrices(a, b, prod, n); // Вызываем функцию умножения

    f = fopen("output.txt", "w"); // Открываем выходной файл для записи результатов
    if (!f) return 1;

    fprintf(f, "Addition (A + B):\n"); // Заголовок суммы
    print_matrix(f, sum, n); // Выводим сумму
    
    fprintf(f, "\nMultiplication (A * B):\n"); // Заголовок умножения
    print_matrix(f, prod, n); // Выводим умножение

    min_val = a[0][0]; // Изначально считаем минимумом самый первый элемент
    min_idx = 1; // Его порядковый номер - 1
    odd_count = 0; // Счетчик нечетных чисел на нуле
    
    for (i = 0; i < n; i++) { // Перебираем строки матрицы A
        for (j = 0; j < n; j++) { // Перебираем столбцы
            cur_idx = i * n + j + 1; // Формула вычисления сквозного порядкового номера (1, 2, 3...)
            if (a[i][j] < min_val) { // Если текущий элемент меньше найденного минимума
                min_val = a[i][j]; // Обновляем минимум
                min_idx = cur_idx; // Запоминаем его новый порядковый номер
            }
            if (a[i][j] % 2 != 0) { // Если остаток от деления на 2 не равен нулю (число нечетное)
                odd_count++; // Увеличиваем счетчик нечетных
            }
        }
    }
    
    fprintf(f, "\nMatrix A stats before swap:\n"); // Выводим собранную статистику
    fprintf(f, "Min element index: %d (Value: %d)\n", min_idx, min_val);
    fprintf(f, "Odd numbers count: %d\n", odd_count);

    swap_pairs(a, n); // Вызываем функцию замены пар местами
    fprintf(f, "\nMatrix A after swapping pairs:\n"); // Выводим измененную матрицу
    print_matrix(f, a, n);

    same_a = has_identical_lines(a, n); // Проверяем матрицу A на одинаковые линии
    same_b = has_identical_lines(b, n); // Проверяем матрицу B
    
    fprintf(f, "\nIdentical lines check:\n"); // Выводим результаты проверки
    fprintf(f, "Matrix A: %s\n", same_a ? "Yes" : "No"); // Если флаг 1, пишем Yes, иначе No
    fprintf(f, "Matrix B: %s\n", same_b ? "Yes" : "No");

    fclose(f); // Закрываем выходной файл

    free_matrix(a, n); // Очищаем память матрицы A
    free_matrix(b, n); // Очищаем B
    free_matrix(sum, n); // Очищаем Sum
    free_matrix(prod, n); // Очищаем Prod

    end_time = clock(); // Стоп секундомер
    work_time = (double)(end_time - start_time) / CLOCKS_PER_SEC; // Вычисляем время в секундах

    printf("Done. Results saved to output.txt\n"); // Сообщение в консоль
    printf("Execution time: %.3f seconds\n", work_time); // Время в консоль

    return 0; // Конец программы
}

        mat[i] = (int*)malloc(n * sizeof(int)); // Память для столбцов
        for (j = 0; j < n; j++) { // Перебор столбцов
            mat[i][j] = rand() % 10; // Случайное число 0-9
        }
    }
    return mat; // Возврат матрицы
}

void free_matrix(int **mat, int n) { // Очистка памяти
    int i; // Счетчик
    if (!mat) return; // Проверка на пустоту
    for (i = 0; i < n; i++) { // Идем по строкам
        free(mat[i]); // Освобождаем каждую
    }
    free(mat); // Освобождаем основу
}

void add_matrices(int **a, int **b, int **res, int n) { // Функция сложения
    int i, j; // Счетчики
    for (i = 0; i < n; i++) { // Строки
        for (j = 0; j < n; j++) { // Столбцы
            res[i][j] = a[i][j] + b[i][j]; // Складываем ячейки с одинаковыми индексами
        }
    }
}

void mult_matrices(int **a, int **b, int **res, int n) { // Функция умножения
    int i, j, k; // Счетчики
    for (i = 0; i < n; i++) { // Строки первой
        for (j = 0; j < n; j++) { // Столбцы второй
            res[i][j] = 0; // Инициализация ячейки нулем
            for (k = 0; k < n; k++) { // Перемножение (строка на столбец)
                res[i][j] += a[i][k] * b[k][j]; // Добавляем к сумме
            }
        }
    }
}

void print_matrix(FILE *f, int **mat, int n) { // Вывод в файл
    int i, j; // Счетчики
    for (i = 0; i < n; i++) { // Строки
        for (j = 0; j < n; j++) { // Столбцы
            fprintf(f, "%4d", mat[i][j]); // Вывод числа
        }
        fprintf(f, "\n"); // Перенос
    }
}

    if (!f) return 1; // Ошибка

    free_matrix(sum, n); // Чистим память суммы
    free_matrix(prod, n); // Чистим память произведения

    return 0; // Конец
}

#include <stdio.h>   // Подключаем ввод/вывод (файлы, консоль)
#include <stdlib.h>  // Подключаем память и генератор
#include <time.h>    // Подключаем таймер

int main(void) { // Главная функция
    int n, m, i, j, k, rows; // Переменные: кол-во элементов (n), столбцов (m), счетчики, кол-во строк (rows)
    int *arr; // Указатель на одномерный массив (вектор)
    int **mat; // Указатель на двумерный массив (матрицу)
    int *r_sums, *c_sums; // Указатели на одномерные массивы сумм строк и сумм столбцов
    FILE *fin, *fout; // Файлы
    clock_t start_time, end_time; // Время
    double work_time; // Разница времени

    start_time = clock(); // Запуск таймера
    srand((unsigned int)time(NULL)); // Настройка рандома

    printf("Enter number of elements N: "); // Запрашиваем размер вектора
    if (scanf("%d", &n) != 1 || n < 1) return 1; // Считываем N
    
    printf("Enter number of columns M: "); // Запрашиваем количество столбцов
    if (scanf("%d", &m) != 1 || m < 1) return 1; // Считываем M

    arr = (int*)malloc(n * sizeof(int)); // Выделяем память под N элементов вектора
    
    fin = fopen("input.txt", "w"); // Открываем на запись
    if (!fin) return 1; // Ошибка
    for (i = 0; i < n; i++) { // Проходим по всем элементам вектора
        arr[i] = rand() % 100 + 1; // Заполняем случайными числами от 1 до 100
        fprintf(fin, "%d ", arr[i]); // Пишем их в файл
    }
    fclose(fin); // Закрываем

    for (i = 0; i < n; i++) { // Имитируем очистку памяти (по заданию)
        arr[i] = 0; // Обнуляем массив
    }

    fin = fopen("input.txt", "r"); // Открываем на чтение
    if (!fin) return 1; // Ошибка
    
    fout = fopen("output.txt", "w"); // Открываем выходной файл на запись
    if (!fout) { // Защита
        fclose(fin); // Если output не открылся, закрываем input
        return 1; // Выходим
    }

    for (i = 0; i < n; i++) { // Считываем из файла N элементов
        fscanf(fin, "%d", &arr[i]); // Читаем число в массив
        if (arr[i] % 35 == 0) { // Если делится и на 5 и на 7 (т.е. на 35) без остатка
            fprintf(fout, "PYATSEM "); // Пишем слово
        } else if (arr[i] % 5 == 0) { // Иначе если делится только на 5
            fprintf(fout, "PYAT "); // Пишем слово
        } else if (arr[i] % 7 == 0) { // Иначе если делится только на 7
            fprintf(fout, "SEM "); // Пишем слово
        } else { // Если ни на что не делится
            fprintf(fout, "%d ", arr[i]); // Выводим само число
        }
    }
    fclose(fin); // Закрываем input
    fclose(fout); // Закрываем output

    rows = n / m; // Вычисляем количество полных строк (целочисленное деление само отбросит остаток)
    
    if (rows > 0) { // Если получается собрать хотя бы одну строку матрицы
        mat = (int**)malloc(rows * sizeof(int*)); // Выделяем память под "костяк" матрицы
        r_sums = (int*)malloc(rows * sizeof(int)); // Выделяем память под массив сумм строк
        c_sums = (int*)malloc(m * sizeof(int)); // Выделяем память под массив сумм столбцов
        
        for (i = 0; i < rows; i++) r_sums[i] = 0; // Изначально обнуляем суммы строк
        for (j = 0; j < m; j++) c_sums[j] = 0; // Изначально обнуляем суммы столбцов

        printf("\nGenerated Matrix %dx%d:\n", rows, m); // Заголовок в консоли
        k = 0; // Счетчик K нужен для прохода по одномерному вектору
        for (i = 0; i < rows; i++) { // Идем по строкам
            mat[i] = (int*)malloc(m * sizeof(int)); // Память под столбцы матрицы
            for (j = 0; j < m; j++) { // Идем по столбцам
                mat[i][j] = arr[k]; // Перекладываем элемент из вектора в матрицу
                k++; // Переходим к следующему элементу вектора
                r_sums[i] += mat[i][j]; // Добавляем число в сумму текущей строки
                c_sums[j] += mat[i][j]; // Добавляем число в сумму текущего столбца
                printf("%4d", mat[i][j]); // Выводим число в консоль
            }
            printf(" | Sum: %d\n", r_sums[i]); // В конце строки выводим её сумму
        }
        
        printf("----------------------\n"); // Разделитель
        for (j = 0; j < m; j++) { // Проходимся по всем столбцам
            printf("%4d", c_sums[j]); // Печатаем массив с суммами столбцов под матрицей
        }
        printf("  <- Column sums\n"); // Подпись

        for (i = 0; i < rows; i++) free(mat[i]); // Чистим строки матрицы
        free(mat); // Чистим "костяк" матрицы
        free(r_sums); // Чистим массив сумм строк
        free(c_sums); // Чистим массив сумм столбцов
    } else { // Если N меньше M (не собрать строку)
        printf("\nNot enough elements to form a matrix.\n"); // Сообщаем об ошибке
    }
    free(arr); // Чистим исходный одномерный вектор

    end_time = clock(); // Остановка таймера
    work_time = (double)(end_time - start_time) / CLOCKS_PER_SEC; // Расчет времени
    
    printf("\nExecution time: %.3f seconds\n", work_time); // Вывод времени

    return 0; // Конец
}
